# Use the .NET 8 base image for Jammy (Ubuntu 22.04)
FROM mcr.microsoft.com/dotnet/aspnet:8.0-jammy AS base

# Install common utilities and dependencies
RUN apt-get update && apt-get install -y \
  fontconfig \
  zip \
  nano \
  curl \
  iputils-ping \
  git \
  python3 \
  python3-pip && \
  apt-get clean && rm -rf /var/lib/apt/lists/*

# Install required Python libraries using pip
RUN python3 -m pip install --upgrade pip && \
  pip install duckdb===1.1.2 && \
  pip install dlt===0.5.4 && \
  pip install pymysql && \
  pip install pyodbc && \
  pip install pg8000 && \
  pip install poetry && \
  pip install pandas===2.2.2 && \
  pip install "dlt[parquet]" && \
  pip install "dlt[filesystem]" && \
  pip install elasticsearch===7.10.0 && \
  pip install numpy===1.26.4 && \
  pip install sqlalchemy oracledb stripe simple-salesforce boto3 pymongo botocore google-analytics-data google-api-python-client google-auth-oauthlib requests_oauthlib facebook-business openpyxl xlrd===2.0.1

# Set working directory inside the container
WORKDIR /application/etl/etlservice

# Expose port 80 for the application
EXPOSE 80
EXPOSE 8080

# Copy application files into the container
COPY application/etl/etlservice .

# Define the entry point for the application
ENTRYPOINT ["dotnet", "BoldDataHub.dll", "--urls=http://0.0.0.0:80"]
