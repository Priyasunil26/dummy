# Use the .NET 8 base image for Noble (Ubuntu 24.04)
FROM mcr.microsoft.com/dotnet/aspnet:8.0-noble AS base

# Install common utilities and dependencies
RUN apt-get update && apt-get install -y \
  fontconfig \
  zip \
  nano \
  curl \
  iputils-ping \
  git \
  python3 \
  python3-pip && \
  apt-get clean && rm -rf /var/lib/apt/lists/*

# Install required Python libraries using pip
RUN pip install duckdb===1.1.2 dlt===0.5.4 pymysql pyodbc pg8000 poetry pandas --break-system-packages
RUN pip install "dlt[parquet]" "dlt[filesystem]" --break-system-packages
RUN pip install sqlalchemy oracledb stripe simple-salesforce boto3 pymongo botocore google-analytics-data google-auth-oauthlib requests_oauthlib facebook-business openpyxl xlrd===2.0.1 --break-system-packages
 
RUN pip install google-api-python-client --break-system-packages
RUN pip install elasticsearch===8.15.1 --break-system-packages

# Set working directory inside the container
WORKDIR /application/etl/etlservice

# Expose port 80 for the application
EXPOSE 80
EXPOSE 8080

# Copy application files into the container
COPY application/etl/etlservice .

# Define the entry point for the application
ENTRYPOINT ["dotnet", "BoldDataHub.dll", "--urls=http://0.0.0.0:80"]
