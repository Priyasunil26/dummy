#See https://aka.ms/containerfastmode to understand how Visual Studio uses this Dockerfile to build your images for faster debugging.

FROM mcr.microsoft.com/dotnet/aspnet:8.0-noble AS base

RUN apt-get update && apt-get install -y \
  fontconfig \
  libgdiplus \
  zip \
  nano \
  curl

RUN apt-get update && \
    apt-get -y install xvfb libasound2t64 libatk1.0-0t64 libc6 libcairo2 libcups2t64 libdbus-1-3 \
	  libexpat1 libfontconfig1 libgbm1 libgcc-s1 libgdk-pixbuf2.0-0 libglib2.0-0t64 libgtk-3-0t64 \
	  libnspr4 libpango-1.0-0 libpangocairo-1.0-0 libstdc++6 libx11-6 libx11-xcb1 libxcb1 libxcomposite1 \
	  libxcursor1 libxdamage1 libxext6 libxfixes3 libxi6 libxrandr2 libxrender1 libxss1 \
	  libxtst6 ca-certificates fonts-liberation libnss3 lsb-release xdg-utils wget \
	&& rm -rf /var/lib/apt/lists/*

WORKDIR /application/reporting/web
EXPOSE 80
COPY application/reporting/web .
ENTRYPOINT ["bash", "entrypoint.sh"]
