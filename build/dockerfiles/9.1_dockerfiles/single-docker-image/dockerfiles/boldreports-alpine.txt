#See https://aka.ms/containerfastmode to understand how Visual Studio uses this Dockerfile to build your images for faster debugging.

FROM mcr.microsoft.com/dotnet/aspnet:8.0-alpine AS base

RUN apk update
RUN apk add curl
RUN apk add bash fontconfig
RUN apk add libgdiplus --repository https://dl-3.alpinelinux.org/alpine/edge/testing/
RUN apk add --no-cache icu-libs krb5-libs libgcc libintl libssl3 libstdc++ zlib tzdata git python3 py3-pip build-base python3-dev libffi-dev

RUN pip install dlt===0.5.4 pymysql pg8000 poetry "dlt[filesystem]" duckdb===1.1.2 pandas --break-system-packages

RUN pip install sqlalchemy oracledb stripe simple-salesforce boto3 pymongo botocore google-analytics-data google-api-python-client google-auth-oauthlib requests_oauthlib facebook-business openpyxl xlrd===2.0.1 --break-system-packages

RUN pip install elasticsearch===8.15.1 --break-system-packages

ENV DOTNET_SYSTEM_GLOBALIZATION_INVARIANT=false
ENV OPENSSL_CONF=/etc/ssl/
ENV OS_ENV=alpine
ENV BOLD_SERVICES_HOSTING_ENVIRONMENT=docker
ENV OPTIONAL_LIBS=mysql,oracle,postgresql,snowflake,googlebigquery

RUN mkdir -p /run/nginx
RUN apk add nginx procps jq

WORKDIR /application

EXPOSE 80
EXPOSE 443

COPY /application .

RUN chmod +x ./entrypoint.sh

CMD ./entrypoint.sh
