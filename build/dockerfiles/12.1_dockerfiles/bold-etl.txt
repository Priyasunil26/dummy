# Use the .NET 8 base image for Noble (Ubuntu 24.04)
FROM mcr.microsoft.com/dotnet/aspnet:8.0-noble AS base

# Install common utilities and dependencies
RUN apt-get update && apt-get install -y \
  fontconfig \
  zip \
  nano \
  curl \
  iputils-ping \
  git \
  python3 \
  python3-pip && \
  apt-get clean && rm -rf /var/lib/apt/lists/*

# Install required Python libraries using pip
RUN pip install duckdb===1.1.2 --break-system-packages
RUN pip install dlt===0.5.4 --break-system-packages
RUN pip install pymysql --break-system-packages
RUN pip install pyodbc --break-system-packages
RUN pip install pg8000 --break-system-packages
RUN pip install poetry --break-system-packages
RUN pip install pandas --break-system-packages
RUN pip install "dlt[parquet]" --break-system-packages
RUN pip install "dlt[filesystem]" --break-system-packages
RUN pip install sqlalchemy --break-system-packages
RUN pip install oracledb --break-system-packages
RUN pip install stripe --break-system-packages
RUN pip install simple-salesforce --break-system-packages
RUN pip install boto3 --break-system-packages
RUN pip install pymongo --break-system-packages
RUN pip install botocore --break-system-packages
RUN pip install google-analytics-data --break-system-packages
RUN pip install google-auth-oauthlib --break-system-packages
RUN pip install requests_oauthlib --break-system-packages
RUN pip install facebook-business --break-system-packages
RUN pip install openpyxl --break-system-packages
RUN pip install xlrd===2.0.1 --break-system-packages
RUN pip install pymssql --break-system-packages

COPY application/etl/etlservice .

# Define the entry point for the application
